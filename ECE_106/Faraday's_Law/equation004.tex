d\vec{s} &= hd\vec{r}(-\hat{y}) \\
\vec{B} &= \frac{\mu_0 I}{2\pi r} (\hat{y}) \quad \text{(From Ampere's Law)} \\ \\

\phi &= \int \vec{B} \cdot d\vec{s} \\
&= \int \frac{\mu_0 I_0 \sin(\omega t)}{2\pi r} \hat{y} \cdot hdr (-\hat{y}) \\
&= \frac{-\mu_0 I_0 \sin(\omega t)}{2\pi r} \int_{r=d}^{d+a} \frac{dr}{r}  \\
&= \frac{-\mu_0 I_0 \sin(\omega t)}{2\pi r} \ln(\frac{d+a}{d})  \\ \\

V_{ind} &= -\frac{d\phi}{dt} \\
V_{ind} &= \frac{-\mu_0 I_0 \omega \cos(\omega t)}{2\pi} \ln(\frac{d+a}{d})  \\
